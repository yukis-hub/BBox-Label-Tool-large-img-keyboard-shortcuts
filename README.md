# General Keyboard shortcuts
"→" = Go Next image  
"←" = Go Previous image  
"left_Ctrl + Z" = Clear previous BBox
"C" = Confirm

# Other Keyboard shortcuts(it depends on class.txt)
You don't need to click Confirm button if using keyboard shortcuts.    
"0-9" = 0-9  
"+" = +  
"-" = -  
"*" = ±  
"i + a" = in_ari  
"i + n" = in_nasi  
"o + a" = out_ari  
"o + n" = out_nasi
